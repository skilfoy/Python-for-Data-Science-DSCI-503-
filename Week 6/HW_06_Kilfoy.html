In [25]:
stl_batting = batting.loc[(batting['teamID'] == 'SLN') & (batting['yearID'] >= 1900)].loc[:, ['yearID', 'H', 'AB', 'HR']]
stl_batting = stl_batting.groupby('yearID').sum()
stl_batting['BA'] = stl_batting['H']/stl_batting['AB']
chi_batting = batting.loc[(batting['teamID'] == 'CHN') & (batting['yearID'] >= 1900)].loc[:, ['yearID', 'H', 'AB', 'HR']]
chi_batting = chi_batting.groupby('yearID').sum()
chi_batting['BA'] = chi_batting['H']/chi_batting['AB']
